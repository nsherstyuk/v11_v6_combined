     1 # Historical Replay Simulator Implementation Plan
     2
     3 > **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowe
       rs:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
     4
     5 **Goal:** Build a replay simulator that feeds historical 1-min bars through the actual live strategy code (Darvas + 4
       H Retest) to verify correctness and measure performance without IBKR.
     6
     7 **Architecture:** ReplayRunner loads historical bars via the existing `data_loader`, creates the same strategy engine
       s as the live system (InstrumentEngine, LevelRetestEngine), and feeds bars directly to them bypassing BarAggregator.
       TradeManager runs in dry_run mode with a stub connection. LLM filter is configurable: passthrough, real Grok, or cach
       ed responses.
     8
     9 **Tech Stack:** Python 3.11+, pytest, asyncio, existing v11 modules (no new dependencies)
    10
    11 **Spec:** `docs/superpowers/specs/2026-04-12-historical-replay-simulator-design.md`

   104 # v11/replay/config.py
   105 """Replay simulator configuration."""
   106 from __future__ import annotations
   107
   108 from dataclasses import dataclass, field
   109 from datetime import datetime
   110
   111 VALID_LLM_MODES = {"passthrough", "live", "cached"}
   112
   113
   114 @dataclass
   115 class ReplayConfig:
   116     """Configuration for a historical replay run."""
   117
   118     # Required
   119     instruments: list[str]
   120     start_date: str              # "YYYY-MM-DD"
   121     end_date: str                # "YYYY-MM-DD"
   122
   123     # LLM mode
   124     llm_mode: str = "passthrough"    # "passthrough" | "live" | "cached"
   125     grok_api_key: str = ""
   126     grok_model: str = "grok-4-1-fast-reasoning"
   127     llm_cache_path: str = "replay_llm_cache.json"
   128     llm_confidence_threshold: int = 75
   129
   130     # Execution simulation
   131     slippage_pips: float = 0.5
   132     commission_per_lot: float = 2.0
   133
   134     # Risk manager
   135     max_daily_loss: float = 500.0
   136     max_daily_trades: int = 10
   137     max_concurrent_positions: int = 3
   138
   139     # Seeding
   140     seed_bars: int = 500             # bars to seed before replay starts
   141
   142     # Output
   143     output_dir: str = "v11/replay/results"
   144     event_verbosity: str = "normal"  # "quiet" | "normal" | "verbose"
   145
   146     # Dry run (always True for replay — no real orders)
   147     dry_run: bool = True
   148
   149     def validate(self) -> None:
   150         """Validate config values. Raises ValueError on problems."""
   151         if not self.instruments:
   152             raise ValueError("instruments must not be empty")
   153         if self.llm_mode not in VALID_LLM_MODES:
   154             raise ValueError(
   155                 f"llm_mode must be one of {VALID_LLM_MODES}, got '{self.llm_mode}'")
   156         start = datetime.strptime(self.start_date, "%Y-%m-%d")
   157         end = datetime.strptime(self.end_date, "%Y-%m-%d")
   158         if start > end:
   159             raise ValueError(
   160                 f"start_date ({self.start_date}) must be before end_date ({self.end_date})")
   161
   162     @property
   163     def start_dt(self) -> datetime:
   164         return datetime.strptime(self.start_date, "%Y-%m-%d")
   165
   166     @property
   167     def end_dt(self) -> datetime:
   168         return datetime.strptime(self.end_date, "%Y-%m-%d")
   169 ```

   432 ```python
   433 # v11/replay/cached_filter.py
   434 """CachedFilter — Record/replay LLM filter responses.
   435
   436 Wraps any LLMFilter implementation with a JSON cache layer.
   437 On cache miss: calls inner filter (if provided), stores response.
   438 On cache hit: returns stored response (instant, free, deterministic).
   439
   440 Cache key: SHA-256 of SignalContext JSON (stable, deterministic).
   441 Cache storage: JSON file mapping hash -> FilterDecision fields.
   442 """
   443 from __future__ import annotations
   444
   445 import hashlib
   446 import json
   447 import logging
   448 from pathlib import Path
   449 from typing import Optional
   450
   451 from ..core.types import FilterDecision
   452 from ..llm.models import SignalContext
   453
   454 log = logging.getLogger(__name__)
   455
   456
   457 class CachedFilter:
   458     """LLM filter with transparent caching.
   459
   460     Satisfies the LLMFilter protocol.
   461     """
   462
   463     def __init__(
   464         self,
   465         inner_filter: Optional[object] = None,
   466         cache_path: str = "replay_llm_cache.json",
   467     ):
   468         self._inner = inner_filter
   469         self._cache_path = Path(cache_path)
   470         self._cache: dict[str, dict] = {}
   471         self._hits = 0
   472         self._misses = 0
   473         self._load_cache()
   474
   475     def _load_cache(self) -> None:
   476         if self._cache_path.exists():
   477             try:
   478                 self._cache = json.loads(self._cache_path.read_text())
   479                 log.info(f"CachedFilter: loaded {len(self._cache)} entries from {self._cache_path}")
   480             except (json.JSONDecodeError, IOError) as e:
   481                 log.warning(f"CachedFilter: failed to load cache: {e}")
   482                 self._cache = {}
   483
   484     def save(self) -> None:
   485         """Persist cache to disk."""
   486         self._cache_path.parent.mkdir(parents=True, exist_ok=True)
   487         self._cache_path.write_text(json.dumps(self._cache, indent=2))
   488         log.info(f"CachedFilter: saved {len(self._cache)} entries "
   489                  f"(hits={self._hits}, misses={self._misses})")
   490
   491     @staticmethod
   492     def _cache_key(context: SignalContext) -> str:
   493         raw = context.model_dump_json(indent=None)
   494         return hashlib.sha256(raw.encode()).hexdigest()[:16]
   495
   496     async def evaluate_signal(self, context: SignalContext) -> FilterDecision:
   497         key = self._cache_key(context)
   498
   499         if key in self._cache:
   500             self._hits += 1
   501             entry = self._cache[key]
   502             log.debug(f"CachedFilter: HIT {key[:8]} -> approved={entry['approved']}")
   503             return FilterDecision(
   504                 approved=entry["approved"],
   505                 confidence=entry["confidence"],
   506                 entry_price=entry["entry_price"],
   507                 stop_price=entry["stop_price"],
   508                 target_price=entry["target_price"],
   509                 reasoning=entry["reasoning"],
   510                 risk_flags=entry.get("risk_flags", []),
   511             )
   512
   513         self._misses += 1
   514
   515         if self._inner is not None:
   516             decision = await self._inner.evaluate_signal(context)
   517             self._cache[key] = {
   518                 "approved": decision.approved,
   519                 "confidence": decision.confidence,
   520                 "entry_price": decision.entry_price,
   521                 "stop_price": decision.stop_price,
   522                 "target_price": decision.target_price,
   523                 "reasoning": decision.reasoning,
   524                 "risk_flags": list(decision.risk_flags),
   525             }
   526             log.debug(f"CachedFilter: MISS {key[:8]} -> called inner, approved={decision.approved}")
   527             return decision
   528
   529         # No inner filter and no cache hit: passthrough
   530         log.debug(f"CachedFilter: MISS {key[:8]} -> no inner filter, passthrough")
   531         return FilterDecision(
   532             approved=True,
   533             confidence=0,
   534             entry_price=context.breakout_price,
   535             stop_price=context.box_bottom,
   536             target_price=0.0,
   537             reasoning="Cache miss — no inner filter, passthrough approval",
   538             risk_flags=["cache_miss"],
   539         )
   540
   541     async def evaluate_orb_signal(self, context) -> FilterDecision:
   542         """ORB not supported in replay — passthrough."""
   543         return FilterDecision(
   544             approved=True, confidence=0,
   545             entry_price=0.0, stop_price=0.0, target_price=0.0,
   546             reasoning="ORB not supported in replay",
   547         )
   548
   549     @property
   550     def stats(self) -> dict:
   551         return {"hits": self._hits, "misses": self._misses, "size": len(self._cache)}
   552 ```

   663 # v11/replay/event_logger.py
   664 """EventLogger — Structured event logging for replay runs.
   665
   666 Emits JSON lines to a file and optionally to console.
   667 Collects trade records for metrics computation.
   668 """
   669 from __future__ import annotations
   670
   671 import json
   672 import logging
   673 from collections import Counter
   674 from pathlib import Path
   675
   676 log = logging.getLogger(__name__)
   677
   678
   679 class EventLogger:
   680     """Structured event logger for replay runs.
   681
   682     Events are written as JSONL (one JSON object per line).
   683     Trade exit events are also collected for metrics.
   684     """
   685
   686     def __init__(
   687         self,
   688         output_path: str,
   689         verbosity: str = "normal",
   690     ):
   691         self._output_path = Path(output_path)
   692         self._output_path.parent.mkdir(parents=True, exist_ok=True)
   693         self._file = open(self._output_path, "w")
   694         self._verbosity = verbosity
   695         self._counts: Counter = Counter()
   696         self._trade_records: list[dict] = []
   697
   698         # Events shown at each verbosity level
   699         self._console_events = {
   700             "quiet": {"TRADE_ENTERED", "TRADE_EXITED", "RISK_LIMIT_HIT"},
   701             "normal": {
   702                 "TRADE_ENTERED", "TRADE_EXITED", "SIGNAL_APPROVED",
   703                 "SIGNAL_REJECTED", "LLM_RESPONSE", "RISK_LIMIT_HIT",
   704                 "DAILY_RESET", "SESSION_START",
   705             },
   706             "verbose": None,  # None = show all
   707         }
   708
   709     def emit(
   710         self,
   711         event: str,
   712         strategy: str,
   713         instrument: str,
   714         timestamp: str,
   715         data: dict,
   716     ) -> None:
   717         """Emit a structured event."""
   718         record = {
   719             "ts": timestamp,
   720             "event": event,
   721             "strategy": strategy,
   722             "instrument": instrument,
   723             "data": data,
   724         }
   725         self._file.write(json.dumps(record, default=str) + "\n")
   726         self._counts[event] += 1
   727
   728         # Collect trade exit records for metrics
   729         if event == "TRADE_EXITED":
   730             self._trade_records.append(data)
   731
   732         # Console output based on verbosity
   733         allowed = self._console_events.get(self._verbosity)
   734         if allowed is None or event in allowed:
   735             self._print_event(record)
   736
   737     def _print_event(self, record: dict) -> None:
   738         event = record["event"]
   739         ts = record["ts"]
   740         inst = record["instrument"]
   741         strategy = record["strategy"]
   742         data = record["data"]
   743
   744         if event == "TRADE_ENTERED":
   745             direction = data.get("direction", "?")
   746             entry = data.get("entry_price", 0)
   747             sl = data.get("stop_price", 0)
   748             tp = data.get("target_price", 0)
   749             print(f"  [{ts}] {inst} {strategy} ENTER {direction} @ {entry} SL={sl} TP={tp}")
   750         elif event == "TRADE_EXITED":
   751             pnl = data.get("pnl", 0)
   752             reason = data.get("exit_reason", "?")
   753             hold = data.get("hold_bars", 0)
   754             print(f"  [{ts}] {inst} {strategy} EXIT {reason} PnL=${pnl:+.2f} hold={hold}bars")
   755         elif event == "SIGNAL_APPROVED":
   756             conf = data.get("confidence", 0)
   757             direction = data.get("direction", "?")
   758             print(f"  [{ts}] {inst} {strategy} APPROVED {direction} conf={conf}")
   759         elif event == "SIGNAL_REJECTED":
   760             reason = data.get("reason", "?")
   761             print(f"  [{ts}] {inst} {strategy} REJECTED: {reason[:80]}")
   762         elif event == "DAILY_RESET":
   763             print(f"  [{ts}] --- DAILY RESET ---")
   764         elif event == "SESSION_START":
   765             bars = data.get("bars", 0)
   766             print(f"  [{ts}] {inst} SESSION START ({bars} bars)")
   767         else:
   768             print(f"  [{ts}] {inst} {strategy} {event}")
   769
   770     @property
   771     def trade_records(self) -> list[dict]:
   772         return self._trade_records
   773
   774     def get_counts(self) -> dict[str, int]:
   775         return dict(self._counts)
   776
   777     def close(self) -> None:
   778         self._file.close()
   779 ```

   863 # v11/replay/metrics.py
   864 """Replay metrics — compute summary statistics from trade records."""
   865 from __future__ import annotations
   866
   867 import math
   868 from typing import List
   869
   870
   871 def compute_metrics(
   872     trades: List[dict],
   873     starting_equity: float = 100_000.0,
   874 ) -> dict:
   875     """Compute summary metrics from a list of trade records.
   876
   877     Each trade record must have at least a 'pnl' key (float).
   878
   879     Returns dict with: total_trades, net_pnl, win_rate, profit_factor,
   880     max_drawdown, max_drawdown_pct, sharpe, avg_pnl, avg_winner, avg_loser.
   881     """
   882     if not trades:
   883         return {
   884             "total_trades": 0, "net_pnl": 0.0, "win_rate": 0.0,
   885             "profit_factor": 0.0, "max_drawdown": 0.0,
   886             "max_drawdown_pct": 0.0, "sharpe": 0.0,
   887             "avg_pnl": 0.0, "avg_winner": 0.0, "avg_loser": 0.0,
   888         }
   889
   890     pnls = [t["pnl"] for t in trades]
   891     winners = [p for p in pnls if p > 0]
   892     losers = [p for p in pnls if p < 0]
   893
   894     gross_profit = sum(winners)
   895     gross_loss = abs(sum(losers))
   896
   897     # Equity curve for drawdown
   898     equity = [starting_equity]
   899     for p in pnls:
   900         equity.append(equity[-1] + p)
   901
   902     peak = equity[0]
   903     max_dd = 0.0
   904     for e in equity[1:]:
   905         if e > peak:
   906             peak = e
   907         dd = peak - e
   908         if dd > max_dd:
   909             max_dd = dd
   910
   911     # Sharpe (annualized, assuming ~252 trading days)
   912     mean_pnl = sum(pnls) / len(pnls)
   913     if len(pnls) > 1:
   914         variance = sum((p - mean_pnl) ** 2 for p in pnls) / (len(pnls) - 1)
   915         std_pnl = math.sqrt(variance)
   916         sharpe = (mean_pnl / std_pnl) * math.sqrt(252) if std_pnl > 0 else 0.0
   917     else:
   918         sharpe = 0.0
   919
   920     return {
   921         "total_trades": len(pnls),
   922         "net_pnl": sum(pnls),
   923         "win_rate": len(winners) / len(pnls) if pnls else 0.0,
   924         "profit_factor": gross_profit / gross_loss if gross_loss > 0 else float("inf"),
   925         "max_drawdown": max_dd,
   926         "max_drawdown_pct": (max_dd / starting_equity) * 100 if starting_equity > 0 else 0.0,
   927         "sharpe": round(sharpe, 2),
   928         "avg_pnl": mean_pnl,
   929         "avg_winner": sum(winners) / len(winners) if winners else 0.0,
   930         "avg_loser": sum(losers) / len(losers) if losers else 0.0,
   931     }
   932 ```

  1169
  1170         # Per-instrument engines (built in run())
  1171         self._engines: Dict[str, List] = {}  # pair -> [engine, ...]
  1172         self._trade_managers: Dict[str, TradeManager] = {}
  1173
  1174     def _build_llm_filter(self):
  1175         """Build LLM filter based on config mode."""
  1176         if self._config.llm_mode == "passthrough":
  1177             return PassthroughFilter()
  1178
  1179         if self._config.llm_mode == "cached":
  1180             inner = None
  1181             if self._config.grok_api_key:
  1182                 from ..llm.grok_filter import GrokFilter
  1183                 inner = GrokFilter(
  1184                     api_key=self._config.grok_api_key,
  1185                     model=self._config.grok_model,
  1186                 )
  1187             return CachedFilter(
  1188                 inner_filter=inner,
  1189                 cache_path=self._config.llm_cache_path,
  1190             )
  1191
  1192         if self._config.llm_mode == "live":
  1193             from ..llm.grok_filter import GrokFilter
  1194             return GrokFilter(
  1195                 api_key=self._config.grok_api_key,
  1196                 model=self._config.grok_model,
  1197             )
  1198
  1199         return PassthroughFilter()
  1200
  1201     def _build_engines(self, instrument: str) -> List:
  1202         """Create Darvas + LevelRetest engines for one instrument."""
  1203         strategy_config = STRATEGY_CONFIGS.get(instrument)
  1204         inst_config = INSTRUMENT_CONFIGS.get(instrument)
  1205         if strategy_config is None or inst_config is None:
  1206             raise ValueError(f"No config for instrument: {instrument}")
  1207
  1208         trade_manager = TradeManager(
  1209             conn=self._conn,
  1210             inst=inst_config,
  1211             log=log,
  1212             trade_log_dir=self._output_dir / "trades",
  1213             dry_run=True,
  1214         )
  1215         self._trade_managers[instrument] = trade_manager
  1216
  1217         darvas = InstrumentEngine(
  1218             strategy_config=strategy_config,
  1219             inst_config=inst_config,
  1220             llm_filter=self._llm_filter,
  1221             trade_manager=trade_manager,
  1222             live_config=self._live_config,
  1223             log=log,
  1224         )
  1225         darvas.strategy_name = "Darvas_Breakout"
  1226         darvas._risk_check = self._risk_manager.can_trade
  1227
  1228         retest = LevelRetestEngine(
  1229             strategy_config=strategy_config,
  1230             inst_config=inst_config,
  1231             llm_filter=self._llm_filter,
  1232             trade_manager=trade_manager,
  1233             live_config=self._live_config,
  1234             log=log,
  1235         )
  1236         retest._risk_check = self._risk_manager.can_trade
  1237
  1238         return [darvas, retest]
  1239
  1240     async def run(self, bars_by_instrument: Dict[str, List[Bar]]) -> dict:
  1241         """Run the full replay.
  1242
  1243         Args:
  1244             bars_by_instrument: {"EURUSD": [Bar, ...], "USDJPY": [Bar, ...]}
  1245
  1246         Returns:
  1247             Summary dict with bars_processed, trade_count, metrics, etc.
  1248         """
  1249         result = {"bars_processed": {}, "trades": [], "metrics": {}}
  1250
  1251         # Build engines for each instrument
  1252         for instrument in self._config.instruments:
  1253             if instrument not in bars_by_instrument:
  1254                 log.warning(f"No bars provided for {instrument}, skipping")
  1255                 continue
  1256             self._engines[instrument] = self._build_engines(instrument)
  1257
  1258         # Process each instrument
  1259         for instrument, bars in bars_by_instrument.items():
  1260             if instrument not in self._engines:
  1261                 continue
  1262
  1263             engines = self._engines[instrument]
  1264             seed_count = min(self._config.seed_bars, len(bars))
  1265             replay_bars = bars[seed_count:]
  1266
  1267             log.info(f"Replay {instrument}: {len(bars)} total bars, "
  1268                      f"{seed_count} seeded, {len(replay_bars)} replayed")
  1269
  1270             # Seed historical bars
  1271             for bar in bars[:seed_count]:
  1272                 for engine in engines:
  1273                     engine.add_historical_bar(bar)
  1274
  1275             self._event_logger.emit(
  1276                 "SESSION_START", strategy="ALL", instrument=instrument,
  1277                 timestamp=replay_bars[0].timestamp.isoformat() if replay_bars else "",
  1278                 data={"total_bars": len(bars), "seed_bars": seed_count,
  1279                       "replay_bars": len(replay_bars)},
  1280             )
  1281
  1282             # Track current date for daily resets
  1283             current_date: Optional[str] = None
  1284
  1285             # Replay loop
  1286             for i, bar in enumerate(replay_bars):
  1287                 bar_date = bar.timestamp.strftime("%Y-%m-%d")
  1288
  1289                 # Daily reset on date change
  1290                 if current_date is not None and bar_date != current_date:
  1291                     self._risk_manager.reset_daily()
  1292                     tm = self._trade_managers[instrument]
  1293                     tm.reset_daily()
  1294                     self._event_logger.emit(
  1295                         "DAILY_RESET", strategy="ALL", instrument=instrument,
  1296                         timestamp=bar.timestamp.isoformat(), data={},
  1297                     )
  1298                 current_date = bar_date
  1299
  1300                 # Feed price to engines (slippage tracking)
  1301                 for engine in engines:
  1302                     engine.on_price(bar.close, bar.timestamp)
  1303
  1304                 # Process the bar through all engines
  1305                 for engine in engines:
  1306                     tm = self._trade_managers[instrument]
  1307                     was_in_trade = tm.in_trade
  1308                     pnl_before = tm.daily_pnl
  1309                     trades_before = tm.daily_trades
  1310
  1311                     await engine.on_bar(bar)
  1312
  1313                     # Detect trade entry
  1314                     if not was_in_trade and tm.in_trade:
  1315                         self._event_logger.emit(
  1316                             "TRADE_ENTERED", strategy=engine.strategy_name,
  1317                             instrument=instrument,
  1318                             timestamp=bar.timestamp.isoformat(),
  1319                             data={
  1320                                 "direction": tm.direction.value if tm.direction else "?",
  1321                                 "entry_price": tm.signal_entry_price,
  1322                                 "stop_price": tm.stop_price,
  1323                                 "target_price": tm.target_price,
  1324                                 "llm_confidence": tm.llm_confidence,
  1325                             },
  1326                         )
  1327
  1328                     # Detect trade exit (daily_trades counter incremented)
  1329                     if tm.daily_trades > trades_before:
  1330                         pnl_delta = tm.daily_pnl - pnl_before
  1331                         self._event_logger.emit(
  1332                             "TRADE_EXITED", strategy=engine.strategy_name,
  1333                             instrument=instrument,
  1334                             timestamp=bar.timestamp.isoformat(),
  1335                             data={
  1336                                 "instrument": instrument,
  1337                                 "strategy": engine.strategy_name,
  1338                                 "pnl": pnl_delta,
  1339                                 "exit_reason": "check_exit",
  1340                                 "hold_bars": 0,
  1341                                 "llm_confidence": 0,
  1342                             },
  1343                         )
  1344
  1345                 # Progress logging every 10000 bars
  1346                 if (i + 1) % 10000 == 0:
  1347                     log.info(f"Replay {instrument}: {i + 1}/{len(replay_bars)} bars")
  1348
  1349             result["bars_processed"][instrument] = len(replay_bars)
  1350
  1351         # Save LLM cache if applicable
  1352         if isinstance(self._llm_filter, CachedFilter):
  1353             self._llm_filter.save()
  1354
  1355         # Compute metrics from trade records
  1356         result["event_counts"] = self._event_logger.get_counts()
  1357         result["trade_records"] = self._event_logger.trade_records
  1358         result["metrics"] = compute_metrics(self._event_logger.trade_records)
  1359
  1360         # Write summary
  1361         self._write_summary(result)
  1362         self._event_logger.close()
  1363
  1364         return result
  1365
  1366     def _write_summary(self, result: dict) -> None:
  1367         """Write human-readable summary file."""
  1368         summary_path = self._output_dir / "replay_summary.txt"
  1369         m = result["metrics"]
  1370         counts = result.get("event_counts", {})
  1371
  1372         lines = [
  1373             f"Replay: {', '.join(self._config.instruments)} "
  1374             f"{self._config.start_date} to {self._config.end_date} "
  1375             f"(LLM: {self._config.llm_mode})",
  1376             "",
  1377         ]
  1378
  1379         for inst, n in result.get("bars_processed", {}).items():
  1380             lines.append(f"  {inst}: {n} bars replayed")
  1381         lines.append("")
  1382
  1383         lines.append(f"Events: {dict(counts)}")
  1384         lines.append("")
  1385
  1386         lines.append("Metrics:")
  1387         lines.append(f"  Total trades: {m['total_trades']}")
  1388         lines.append(f"  Net PnL: ${m['net_pnl']:+.2f}")
  1389         lines.append(f"  Win rate: {m['win_rate']:.1%}")
  1390         lines.append(f"  Profit factor: {m['profit_factor']:.2f}")
  1391         lines.append(f"  Sharpe: {m['sharpe']:.2f}")
  1392         lines.append(f"  Max drawdown: ${m['max_drawdown']:.2f} ({m['max_drawdown_pct']:.1f}%)")
  1393         lines.append(f"  Avg winner: ${m['avg_winner']:+.2f}")
  1394         lines.append(f"  Avg loser: ${m['avg_loser']:+.2f}")
  1395
  1396         summary_path.write_text("\n".join(lines))
  1397         log.info(f"Summary written to {summary_path}")
  1398         print("\n" + "\n".join(lines))
